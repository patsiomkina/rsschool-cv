def is_isogram(string):
    new_str = ''
    for letter in string:
        new_str += letter
    new_set = set(new_str.lower())
    finish = ''.join(sorted(new_set))
    start = ''.join(sorted(string.lower()))
    if start == finish or string == '':
        return True
    else:
        return False
